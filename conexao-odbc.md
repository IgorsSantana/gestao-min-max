# Consulta ODBC - Vendas do Período e Comprador Principal

Esta é a consulta SQL utilizada para obter as vendas do período e informações do comprador principal via conexão ODBC.

## Dados de Conexão (ODBC)

**String de Conexão:**
```ini
DRIVER={IBM DB2 ODBC DRIVER};DATABASE=SAB;HOSTNAME=10.64.1.11;PORT=50000;PROTOCOL=TCPIP;UID=db2user_ro;PWD=Sup3rs4nt0;
```

**Parâmetros detalhados:**
- **Driver:** IBM DB2 ODBC DRIVER
- **Database:** SAB
- **Host (IP):** 10.64.1.11
- **Porta:** 50000
- **Protocolo:** TCPIP
- **Usuário:** db2user_ro
- **Senha:** Sup3rs4nt0

## SQL

```sql
select
    EA.IDEMPRESA as EMPRESA,
    SC.IDSECAO as SECAO,
    SC.DESCRSECAO as DESCRICAO_SECAO,
    GR.IDGRUPO as GRUPO,
    GR.DESCRGRUPO as DESCRICAO_GRUPO,
    SG.IDSUBGRUPO as SUBGRUPO,
    SG.DESCRSUBGRUPO as DESCRICAO_SUBGRUPO,
    EA.IDSUBPRODUTO as COD_INTERNO,
    PG.IDCODBARPROD as COD_EAN,
    PR.FABRICANTE as CURVA,
    US.NOMEUSUARIO as COMPRADOR,
    PR.DESCRCOMPRODUTO || ' ' || PG.SUBDESCRICAO AS DESCRICAO_PRODUTO,
    PP.CUSTOGERENCIAL as CUSTO_GERENC,
    PP.VALPRECOVAREJO as PRECO_VAREJO,
    CASE WHEN DATE(PG.DTCADASTRO) <= '2025-12-31' THEN 'N' ELSE 'S' END AS CADASTRO_NOVO,
    CASE WHEN DATE(PG.DTCADASTRO) <= '2025-12-31' THEN COALESCE(PC.QTDESTMINIMO, 0) ELSE 0 END AS ESTOQUE_MINIMO_ATUAL,
    CASE WHEN DATE(PG.DTCADASTRO) <= '2025-12-31' THEN COALESCE(PC.QTDESTMAXIMO, 0) ELSE 0 END AS ESTOQUE_MAXIMO_ATUAL,
    CASE WHEN DATE(PG.DTCADASTRO) >= '2026-01-01' THEN COALESCE(PC.QTDESTMINIMO, 0) ELSE 0 END AS ESTOQUE_MINIMO_NOVO,
    CASE WHEN DATE(PG.DTCADASTRO) >= '2026-01-01' THEN COALESCE(PC.QTDESTMAXIMO, 0) ELSE 0 END AS ESTOQUE_MAXIMO_NOVO,
    PC.DIASESTOQUE as  DIAS_ESTOQUE,
    count( distinct(EA.DTMOVIMENTO) ) as DIAS_VENDA,
    sum(EA.QUANT_VENDA) as QTDE_VENDA,
    min(EA.DTMOVIMENTO) as PRIM_VENDA,
    max(EA.DTMOVIMENTO) as ULTI_VENDA
from
    (
    select
        A.IDEMPRESA,
        A.DTMOVIMENTO,
        A.IDPRODUTO,
        A.IDSUBPRODUTO,
        cast( sum(A.QTDPRODUTO) as decimal (15,3) ) as QUANT_VENDA
    from
        DBA.ESTOQUE_ANALITICO A
    where
        A.FLAGMOVSALDOPRO = 'T' and
        A.TIPOCATEGORIA = 'A' and
        A.IDOPERACAO > 1000 and
        A.IDOPERACAO <> 1301 and
        A.DTMOVIMENTO between ((CURRENT DATE - 1 DAY) - 1 YEAR) and (CURRENT DATE - 1 DAY)
    group by
        A.IDEMPRESA,
        A.DTMOVIMENTO,
        A.IDPRODUTO,
        A.IDSUBPRODUTO
    ) as EA
    left join DBA.PRODUTO_COMPRAS PC on (EA.IDEMPRESA = PC.IDEMPRESA and EA.IDPRODUTO = PC.IDPRODUTO and EA.IDSUBPRODUTO = PC.IDSUBPRODUTO),
    DBA.POLITICA_PRECO_PRODUTO PP,
    DBA.PRODUTO PR
    left join DBA.SECAO SC on (PR.IDSECAO = SC.IDSECAO)
    left join DBA.GRUPO GR on (PR.IDGRUPO = GR.IDGRUPO)
    left join DBA.SUBGRUPO SG on (PR.IDSUBGRUPO = SG.IDSUBGRUPO)
    left join DBA.PRODUTO_GRADE PG on (PR.IDPRODUTO = PG.IDPRODUTO)
    left join
    (
        select
            min(IDUSUARIO) as IDUSUARIO,
            IDPRODUTO,
            IDSUBPRODUTO
        from
            DBA.PRODUTO_COMPRADOR
        group by
            IDPRODUTO, IDSUBPRODUTO
    ) as CP on (CP.IDPRODUTO = PG.IDPRODUTO and CP.IDSUBPRODUTO = PG.IDSUBPRODUTO)
    left join DBA.USUARIO US on (CP.IDUSUARIO = US.IDUSUARIO)
where
    PG.IDPRODUTO = EA.IDPRODUTO and
    PG.IDSUBPRODUTO = EA.IDSUBPRODUTO and
    PP.IDPRODUTO = PG.IDPRODUTO and
    PP.IDSUBPRODUTO = PG.IDSUBPRODUTO and
    PP.IDEMPRESA = EA.IDEMPRESA and
    PG.FLAGINATIVO = 'F' and
    PG.FLAGINATIVOCOMPRA = 'F'
group by
    EA.IDEMPRESA,
    SC.IDSECAO,
    SC.DESCRSECAO,
    GR.IDGRUPO,
    GR.DESCRGRUPO,
    SG.IDSUBGRUPO,
    SG.DESCRSUBGRUPO,
    EA.IDPRODUTO,
    EA.IDSUBPRODUTO,
    PG.IDCODBARPROD,
    PR.FABRICANTE,
    US.NOMEUSUARIO,
    PR.DESCRCOMPRODUTO || ' ' || PG.SUBDESCRICAO,
    PP.CUSTOGERENCIAL,
    PP.VALPRECOVAREJO,
    PG.DTCADASTRO,
    PC.QTDESTMINIMO,
    PC.QTDESTMAXIMO,
    PC.DIASESTOQUE
order by
    SC.IDSECAO,
    GR.IDGRUPO,
    SG.IDSUBGRUPO,
    DESCRICAO_PRODUTO,
    EA.IDEMPRESA

```
